465 void push(detail::node_ptr node, std::byte key_byte,
466 std::uint8_t child_index, detail::key_prefix_snapshot prefix) {
467 // For variable length keys we need to know the number of bytes associated
468 // with the node's key_prefix. In addition there is one byte for the
469 // descent to the child node along the child_index. That information needs
470 // to be stored on the stack so we can pop off the right number of bytes
471 // even for OLC where the node might be concurrently modified.
472 UNODB_DETAIL_ASSERT(node.type() != node_type::LEAF);
473 stack_.push({node, key_byte, child_index, prefix});
474 keybuf_.push(prefix.get_key_view());
475 keybuf_.push(key_byte);
476 }

481 void push_leaf(detail::node_ptr aleaf) {
482 // Mock up a stack entry for the leaf.
483 stack_.push({
484 aleaf,
485 static_cast<std::byte>(0xFFU), // ignored for leaf
486 static_cast<std::uint8_t>(0xFFU), // ignored for leaf
487 detail::key_prefix_snapshot(0) // ignored for leaf
488 });
489 // No change in the key_buffer.
490 }

493 void push(const typename inode_base::iter_result& e) {
494 const auto node_type = e.node.type();
495 if (UNODB_DETAIL_UNLIKELY(node_type == node_type::LEAF)) {
496 push_leaf(e.node);
497 }
498 push(e.node, e.key_byte, e.child_index, e.prefix);
499 }

1474 struct stack_entry {
1475 detail::node_ptr* slot;

1477 };
1478
1479 boost::container::small_vector<stack_entry, 32> stack;
1480 auto* slot = &root;
1481 auto remaining_key{remove_key};
1482
1483 // --- Downward pass: find the leaf ---
1484 while (true) {
1485 const auto node_val = *slot;
1486 const auto ntype = node_val.type();

1488
1489 auto* const inode{node_val.template ptr<inode_type*>()};
1490 const auto& kp{inode->get_key_prefix()};
1491 const auto kp_len{kp.length()};
1492 if (kp.get_shared_length(remaining_key) < kp_len) return false;
1493 remaining_key.shift_right(kp_len);
1494
1495 const auto [child_i, child_ptr]{inode->find_child(ntype, remaining_key[0])};
1496 if (child_ptr == nullptr) return false;
1497
1498 const auto child_val{child_ptr->load()};
1499 if (child_val.type() != node_type::LEAF) {
1500 stack.push_back({slot, child_i});
1501 slot = detail::unwrap_fake_critical_section(child_ptr);
1502 remaining_key.shift_right(1);
1503 continue;
1504 }
1505
1506 // Found a leaf — verify it matches.
1507 auto* const leaf{child_val.template ptr<leaf_type*>()};

1509
1510 // --- Upward pass ---
1511 const auto count = inode->get_children_count();
1512
1513 if (count > 1 || ntype != node_type::I4) {
1514 const auto remove_result
1515 UNODB_DETAIL_USED_IN_DEBUG{inode->template remove_or_choose_subtree<

1523 // Single-child inode (chain node). Reclaim leaf and chain,
1524 // then walk up cleaning any further empty chains.
1525 {
1526 const auto rl{art_policy::reclaim_leaf_on_scope_exit(leaf, *this)};
1527 }

1648 UNODB_DETAIL_ASSERT(node != nullptr);
1649 const auto node_type = node.type();

1665 push(e2);

1701template <typename Key, typename Value>
1702typename db<Key, Value>::iterator&
1703db<Key, Value>::iterator::left_most_traversal(detail::node_ptr node) {
1704 while (true) {
1705 UNODB_DETAIL_ASSERT(node != nullptr);
1706 const auto node_type = node.type();
1707 if (node_type == node_type::LEAF) {
1708 push_leaf(node);
1709 return *this; // done
1710 }
1711 // recursive descent.
1712 auto* const inode{node.ptr<inode_type*>()};
1713 const auto e =
1714 inode->begin(node_type); // first child of current internal node
1715 push(e); // push the entry on the stack.
1716 node = inode->get_child(node_type, e.child_index); // get the child
1717 }
1718 UNODB_DETAIL_CANNOT_HAPPEN();

1728 push_leaf(node);
1729 return *this; // done
1730 }

1732 auto* const inode{node.ptr<inode_type*>()};

1734 inode->last(node_type); // first child of current internal node
1735 push(e); // push the entry on the stack.

1752 while (true) {
1753 const auto node_type = node.type();
1754 if (node_type == node_type::LEAF) {
1755 const auto* const leaf{node.template ptr<leaf_type*>()};
1756 push_leaf(node);
1757 const auto cmp_ = leaf->cmp(k);
1758 if (cmp_ == 0) {
1759 match = true;
1760 return *this;
1761 }
1762 if (fwd) { // GTE semantics
1763 // if search_key < leaf, use the leaf, else next().
1764 return (cmp_ < 0) ? *this : next();
1765 }
1766 // LTE semantics: if search_key > leaf, use the leaf, else prior().
1767 return (cmp_ > 0) ? *this : prior();
1768 }
1769 UNODB_DETAIL_ASSERT(node_type != node_type::LEAF);
1770 auto* const inode{node.template ptr<inode_type*>()}; // some internal node.
1771 const auto key_prefix{inode->get_key_prefix().get_snapshot()}; // prefix
1772 const auto key_prefix_length{key_prefix.length()}; // length of that prefix
1773 const auto shared_length = key_prefix.get_shared_length(
1774 remaining_key.get_u64()); // #of prefix bytes matched.
1775 if (shared_length < key_prefix_length) {
1776 // We have visited an internal node whose prefix is longer than
1777 // the bytes in the key that we need to match. To figure out
1778 // whether the search key would be located before or after the
1779 // current internal node, we need to compare the respective key
1780 // spans lexicographically. Since we have [shared_length] bytes
1781 // in common, we know that the next byte will tell us the
1782 // relative ordering of the key vs the prefix. So now we compare
1783 // prefix and key and the first byte where they differ.
1784 const auto cmp_ = static_cast<int>(remaining_key[shared_length]) -
1785 static_cast<int>(key_prefix[shared_length]);
1786 UNODB_DETAIL_ASSERT(cmp_ != 0);
1787 if (fwd) {
1788 if (cmp_ < 0) {
1789 // FWD and the search key is ordered before this node. We
1790 // want the left-most leaf under the node.
1791 return left_most_traversal(node);
1792 }
1793 // FWD and the search key is ordered after this node. Right
1794 // most descent and then next().
1795 return right_most_traversal(node).next();
1796 }
1797 // reverse traversal
1798 if (cmp_ < 0) {
1799 // REV and the search key is ordered before this node. We
1800 // want the preceding key.
1801 return left_most_traversal(node).prior();
1802 }
1803 // REV and the search key is ordered after this node.
1804 return right_most_traversal(node);
1805 }
1806 remaining_key.shift_right(key_prefix_length);
1807 const auto res = inode->find_child(node_type, remaining_key[0]);
1808 if (res.second == nullptr) {
1809 // We are on a key byte during the descent that is not mapped by
1810 // the current node. Where we go next depends on whether we are
1811 // doing forward or reverse traversal.
1812 if (fwd) {
1813 // FWD: Take the next child_index that is mapped in the data
1814 // and then do a left-most descent to land on the key that is
1815 // the immediate successor of the desired key in the data.
1816 //
1817 // Note: We are probing with a key byte which does not appear
1818 // in our list of keys (this was verified above) so this will
1819 // always be the index the first entry whose key byte is
1820 // greater-than the probe value and [false] if there is no
1821 // such entry.
1822 //
1823 // Note: [node] has not been pushed onto the stack yet!
1824 auto nxt = inode->gte_key_byte(node_type, remaining_key[0]);
1825 if (!nxt) {
1826 // Pop entries off the stack until we find one with a
1827 // right-sibling of the path we took to this node and then
1828 // do a left-most descent under that right-sibling. If there
1829 // is no such parent, we will wind up with an empty stack
1830 // (iterator at the end) and return that state.
1831 if (!empty()) pop();
1832 while (!empty()) {
1833 const auto& centry = top();
1834 const auto cnode{centry.node}; // possible parent from the stack
1835 auto* const icnode{cnode.template ptr<inode_type*>()};
1836 const auto cnxt = icnode->next(
1837 cnode.type(), centry.child_index); // right-sibling.
1838 if (cnxt) {
1839 auto nchild = icnode->get_child(cnode.type(), centry.child_index);
1840 return left_most_traversal(nchild);
1841 }
1842 pop();
1843 }
1844 return *this; // stack is empty (aka end iterator).
1845 }
1846 const auto& tmp = nxt.value(); // unwrap.
1847 const auto child_index = tmp.child_index;
1848 const auto child = inode->get_child(node_type, child_index);
1849 push(node, tmp.key_byte, child_index, tmp.prefix); // the path we took
1850 return left_most_traversal(child); // left most traversal
1851 }
1852 // REV: Take the prior child_index that is mapped and then do
1853 // a right-most descent to land on the key that is the
1854 // immediate predecessor of the desired key in the data.
1855 auto nxt = inode->lte_key_byte(node_type, remaining_key[0]);
1856 if (!nxt) {
1857 // Pop off the current entry until we find one with a
1858 // left-sibling and then do a right-most descent under that
1859 // left-sibling. In the extreme case there is no such
1860 // previous entry and we will wind up with an empty stack.
1861 if (!empty()) pop();
1862 while (!empty()) {
1863 const auto& centry = top();
1864 const auto cnode{centry.node}; // possible parent from stack
1865 auto* const icnode{cnode.template ptr<inode_type*>()};
1866 const auto cnxt =
1867 icnode->prior(cnode.type(), centry.child_index); // left-sibling.
1868 if (cnxt) {
1869 auto nchild = icnode->get_child(cnode.type(), centry.child_index);
1870 return right_most_traversal(nchild);
1871 }
1872 pop();
1873 }
1874 return *this; // stack is empty (aka end iterator).
1875 }
1876 const auto& tmp = nxt.value(); // unwrap.
1877 const auto child_index{tmp.child_index};
1878 const auto child = inode->get_child(node_type, child_index);
1879 push(node, tmp.key_byte, child_index, tmp.prefix); // the path we took
1880 return right_most_traversal(child); // right most traversal
1881 }
1882 // Simple case. There is a child for the current key byte.
1883 const auto child_index{res.first};
1884 const auto* const child{res.second};
1885 push(node, remaining_key[0], child_index, key_prefix);
1886 node = *child;
1887 remaining_key.shift_right(1);
1888 } // while ( true )
1889 UNODB_DETAIL_CANNOT_HAPPEN();
1890}
