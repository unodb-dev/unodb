449 void push(detail::node_ptr node, std::byte key_byte,
450 std::uint8_t child_index, detail::key_prefix_snapshot prefix) {
451 // For variable length keys we need to know the number of bytes associated
452 // with the node's key_prefix. In addition there is one byte for the
453 // descent to the child node along the child_index. That information needs
454 // to be stored on the stack so we can pop off the right number of bytes
455 // even for OLC where the node might be concurrently modified.
456 UNODB_DETAIL_ASSERT(node.type() != node_type::LEAF);
457 stack_.push({node, key_byte, child_index, prefix});
458 keybuf_.push(prefix.get_key_view());
459 keybuf_.push(key_byte);
460 }

465 void push_leaf(detail::node_ptr aleaf) {
466 // Mock up a stack entry for the leaf.
467 stack_.push({
468 aleaf,
469 static_cast<std::byte>(0xFFU), // ignored for leaf
470 static_cast<std::uint8_t>(0xFFU), // ignored for leaf
471 detail::key_prefix_snapshot(0) // ignored for leaf
472 });
473 // No change in the key_buffer.
474 }

477 void push(const typename inode_base::iter_result& e) {
478 const auto node_type = e.node.type();
479 if (UNODB_DETAIL_UNLIKELY(node_type == node_type::LEAF)) {

483 }

1137 db<Key, Value>& db_instance, tree_depth<basic_art_key<Key>> depth,
1138 detail::node_ptr* node_in_parent) {

1140 unwrap_fake_critical_section(inode.find_child(key_byte).second);
1141
1142 if (child != nullptr) return child;

1144 auto leaf = art_policy<Key, Value>::make_db_leaf_ptr(k, v, db_instance);
1145 const auto children_count = inode.get_children_count();
1146
1147 if constexpr (!std::is_same_v<INode, inode_256<Key, Value>>) {
1148 if (UNODB_DETAIL_UNLIKELY(children_count == INode::capacity)) {
1149 auto larger_node{INode::larger_derived_type::create(
1150 db_instance, inode, std::move(leaf), depth)};

1152 node_ptr{larger_node.release(), INode::larger_derived_type::type};

1154 db_instance
1155 .template account_growing_inode<INode::larger_derived_type::type>();
1156#endif // UNODB_DETAIL_WITH_STATS
1157 return child;
1158 }
1159 }
1160 inode.add_to_nonfull(std::move(leaf), depth, children_count);
1161 return child;

1274 // new inode as its direct children.

1395template <typename Key, typename Value>
1396typename db<Key, Value>::iterator& db<Key, Value>::iterator::next() {
1397 while (!empty()) {
1398 const auto& e = top();
1399 const auto node{e.node};
1400 UNODB_DETAIL_ASSERT(node != nullptr);
1401 const auto node_type = node.type();
1402 if (node_type == node_type::LEAF) {
1403 pop(); // pop off the leaf
1404 continue; // falls through loop if just a root leaf since stack now
1405 // empty.
1406 }
1407 auto* inode{node.template ptr<inode_type*>()};
1408 const auto nxt = inode->next(node_type,
1409 e.child_index); // next child of that parent.
1410 if (!nxt.has_value()) {
1411 pop(); // Nothing more for that inode.
1412 continue; // We will look for the right sibling of the parent inode.
1413 }
1414 // Fix up stack for new parent node state and left-most descent.
1415 const auto& e2 = nxt.value();
1416 pop();
1417 push(e2);
1418 const auto child = inode->get_child(node_type, e2.child_index); // descend
1419 return left_most_traversal(child);
1420 }
1421 return *this; // stack is empty, so iterator is at the end
1422}

1428 const auto node{e.node};

1430 const auto node_type = node.type();

1445 pop();
1446 push(e2);
1447 auto child = inode->get_child(node_type, e2.child_index); // descend
1448 return right_most_traversal(child);
1449 }
1450 return *this; // stack is empty, so iterator is at the end.

1456 while (true) {
1457 UNODB_DETAIL_ASSERT(node != nullptr);
1458 const auto node_type = node.type();

1460 push_leaf(node);

1466 inode->begin(node_type); // first child of current internal node
1467 push(e); // push the entry on the stack.
1468 node = inode->get_child(node_type, e.child_index); // get the child
1469 }

1493template <typename Key, typename Value>
1494typename db<Key, Value>::iterator& db<Key, Value>::iterator::seek(
1495 art_key_type search_key, bool& match, bool fwd) {
1496 invalidate(); // invalidate the iterator (clear the stack).
1497 match = false; // unless we wind up with an exact match.
1498 if (UNODB_DETAIL_UNLIKELY(db_.root == nullptr)) return *this; // aka end
1499
1500 auto node{db_.root};
1501 const auto k = search_key;
1502 auto remaining_key{k};
1503
1504 while (true) {
1505 const auto node_type = node.type();
1506 if (node_type == node_type::LEAF) {
1507 const auto* const leaf{node.template ptr<leaf_type*>()};
1508 push_leaf(node);
1509 const auto cmp_ = leaf->cmp(k);
1510 if (cmp_ == 0) {
1511 match = true;
1512 return *this;
1513 }
1514 if (fwd) { // GTE semantics
1515 // if search_key < leaf, use the leaf, else next().
1516 return (cmp_ < 0) ? *this : next();
1517 }
1518 // LTE semantics: if search_key > leaf, use the leaf, else prior().
1519 return (cmp_ > 0) ? *this : prior();
1520 }
1521 UNODB_DETAIL_ASSERT(node_type != node_type::LEAF);
1522 auto* const inode{node.template ptr<inode_type*>()}; // some internal node.
1523 const auto key_prefix{inode->get_key_prefix().get_snapshot()}; // prefix
1524 const auto key_prefix_length{key_prefix.length()}; // length of that prefix
1525 const auto shared_length = key_prefix.get_shared_length(
1526 remaining_key.get_u64()); // #of prefix bytes matched.
1527 if (shared_length < key_prefix_length) {
1528 // We have visited an internal node whose prefix is longer than
1529 // the bytes in the key that we need to match. To figure out
1530 // whether the search key would be located before or after the
1531 // current internal node, we need to compare the respective key
1532 // spans lexicographically. Since we have [shared_length] bytes
1533 // in common, we know that the next byte will tell us the
1534 // relative ordering of the key vs the prefix. So now we compare
1535 // prefix and key and the first byte where they differ.
1536 const auto cmp_ = static_cast<int>(remaining_key[shared_length]) -
1537 static_cast<int>(key_prefix[shared_length]);
1538 UNODB_DETAIL_ASSERT(cmp_ != 0);
1539 if (fwd) {
1540 if (cmp_ < 0) {
1541 // FWD and the search key is ordered before this node. We
1542 // want the left-most leaf under the node.
1543 return left_most_traversal(node);
1544 }
1545 // FWD and the search key is ordered after this node. Right
1546 // most descent and then next().
1547 return right_most_traversal(node).next();
1548 }
1549 // reverse traversal
1550 if (cmp_ < 0) {
1551 // REV and the search key is ordered before this node. We
1552 // want the preceding key.
1553 return left_most_traversal(node).prior();
1554 }
1555 // REV and the search key is ordered after this node.
1556 return right_most_traversal(node);
1557 }
1558 remaining_key.shift_right(key_prefix_length);
1559 const auto res = inode->find_child(node_type, remaining_key[0]);
1560 if (res.second == nullptr) {
1561 // We are on a key byte during the descent that is not mapped by
1562 // the current node. Where we go next depends on whether we are
1563 // doing forward or reverse traversal.
1564 if (fwd) {
1565 // FWD: Take the next child_index that is mapped in the data
1566 // and then do a left-most descent to land on the key that is
1567 // the immediate successor of the desired key in the data.
1568 //
1569 // Note: We are probing with a key byte which does not appear
1570 // in our list of keys (this was verified above) so this will
1571 // always be the index the first entry whose key byte is
1572 // greater-than the probe value and [false] if there is no
1573 // such entry.
1574 //
1575 // Note: [node] has not been pushed onto the stack yet!
1576 auto nxt = inode->gte_key_byte(node_type, remaining_key[0]);
1577 if (!nxt) {
1578 // Pop entries off the stack until we find one with a
1579 // right-sibling of the path we took to this node and then
1580 // do a left-most descent under that right-sibling. If there
1581 // is no such parent, we will wind up with an empty stack
1582 // (iterator at the end) and return that state.
1583 if (!empty()) pop();
1584 while (!empty()) {
1585 const auto& centry = top();
1586 const auto cnode{centry.node}; // possible parent from the stack
1587 auto* const icnode{cnode.template ptr<inode_type*>()};
1588 const auto cnxt = icnode->next(
1589 cnode.type(), centry.child_index); // right-sibling.
1590 if (cnxt) {
1591 auto nchild = icnode->get_child(cnode.type(), centry.child_index);
1592 return left_most_traversal(nchild);
1593 }
1594 pop();
1595 }
1596 return *this; // stack is empty (aka end iterator).
1597 }
1598 const auto& tmp = nxt.value(); // unwrap.
1599 const auto child_index = tmp.child_index;
1600 const auto child = inode->get_child(node_type, child_index);
1601 push(node, tmp.key_byte, child_index, tmp.prefix); // the path we took
1602 return left_most_traversal(child); // left most traversal
1603 }
1604 // REV: Take the prior child_index that is mapped and then do
1605 // a right-most descent to land on the key that is the
1606 // immediate predecessor of the desired key in the data.
1607 auto nxt = inode->lte_key_byte(node_type, remaining_key[0]);
1608 if (!nxt) {
1609 // Pop off the current entry until we find one with a
1610 // left-sibling and then do a right-most descent under that
1611 // left-sibling. In the extreme case there is no such
1612 // previous entry and we will wind up with an empty stack.
1613 if (!empty()) pop();
1614 while (!empty()) {
1615 const auto& centry = top();
1616 const auto cnode{centry.node}; // possible parent from stack
1617 auto* const icnode{cnode.template ptr<inode_type*>()};
1618 const auto cnxt =
1619 icnode->prior(cnode.type(), centry.child_index); // left-sibling.
1620 if (cnxt) {
1621 auto nchild = icnode->get_child(cnode.type(), centry.child_index);
1622 return right_most_traversal(nchild);
1623 }
1624 pop();
1625 }
1626 return *this; // stack is empty (aka end iterator).
1627 }
1628 const auto& tmp = nxt.value(); // unwrap.
1629 const auto child_index{tmp.child_index};
1630 const auto child = inode->get_child(node_type, child_index);
1631 push(node, tmp.key_byte, child_index, tmp.prefix); // the path we took
1632 return right_most_traversal(child); // right most traversal
1633 }
1634 // Simple case. There is a child for the current key byte.
1635 const auto child_index{res.first};
1636 const auto* const child{res.second};
1637 push(node, remaining_key[0], child_index, key_prefix);
1638 node = *child;
1639 remaining_key.shift_right(1);
1640 } // while ( true )
1641 UNODB_DETAIL_CANNOT_HAPPEN();
1642}

1722template <typename Key, typename Value>
1723template <typename FN>
1724void db<Key, Value>::scan_range(Key from_key, Key to_key, FN fn) {
1725 // TODO(thompsonbry) : variable length keys. Explore a cheaper way
1726 // to handle the exclusive bound case when developing variable
1727 // length key support based on the maintained key buffer.
1728 constexpr bool debug = false; // set true to debug scan.
1729 const art_key_type from_key_{from_key}; // convert to internal key
1730 const art_key_type to_key_{to_key}; // convert to internal key
1731 const auto ret = from_key_.cmp(to_key_); // compare the internal keys
1732 const bool fwd{ret < 0}; // from_key is less than to_key
1733 if (ret == 0) return; // NOP
1734 bool match{};
1735 if (fwd) {
1736 iterator it(*this);
1737 it.seek(from_key_, match, true /*fwd*/);
1738 if constexpr (debug) {
1739 std::cerr << "scan_range:: fwd"
1740 << ", from_key=" << from_key_ << ", to_key=" << to_key_ << "\n";
1741 it.dump(std::cerr);
1742 }
1743 const visitor_type v{it};
1744 while (it.valid() && it.cmp(to_key_) < 0) {
1745 if (UNODB_DETAIL_UNLIKELY(fn(v))) break;
1746 it.next();
1747 if constexpr (debug) {
1748 std::cerr << "scan_range:: next()\n";
1749 it.dump(std::cerr);
1750 }
1751 }
1752 } else { // reverse traversal.
1753 iterator it(*this);
