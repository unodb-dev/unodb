508 [[nodiscard]] int cmp(art_key_type akey) const noexcept {
509 UNODB_DETAIL_ASSERT(!stack_.empty());
510 if constexpr (art_policy::full_key_in_inode_path) {
511 return unodb::detail::compare(keybuf_.get_key_view(),
512 akey.get_key_view());

514 auto& node = stack_.top().node;
515 UNODB_DETAIL_ASSERT(node.type() == node_type::LEAF);
516 const auto* const leaf{node.template ptr<leaf_type*>()};
517 return unodb::detail::compare(leaf->get_key_view(),

537 void push(detail::node_ptr node, std::byte key_byte,
538 std::uint8_t child_index, detail::key_prefix_snapshot prefix) {
539 // For variable length keys we need to know the number of bytes associated
540 // with the node's key_prefix. In addition there is one byte for the
541 // descent to the child node along the child_index. That information needs
542 // to be stored on the stack so we can pop off the right number of bytes
543 // even for OLC where the node might be concurrently modified.
544 UNODB_DETAIL_ASSERT(node.type() != node_type::LEAF);
545 stack_.push({node, key_byte, child_index, prefix});
546 keybuf_.push(prefix.get_key_view());
547 keybuf_.push(key_byte);
548 }
549
553 void push_leaf(detail::node_ptr aleaf) {
554 stack_.push({
555 aleaf,
556 static_cast<std::byte>(0xFFU), // ignored for leaf
557 static_cast<std::uint8_t>(0xFFU), // ignored for leaf
558 detail::key_prefix_snapshot(0), // ignored for leaf
559 true // packed_leaf
560 });
561 // No change in the key_buffer.
562 }
563
565 void push(const typename inode_base::iter_result& e) {
566 const auto node_type = e.node.type();
567 if (UNODB_DETAIL_UNLIKELY(node_type == node_type::LEAF)) {
568 push_leaf(e.node);
569 return;
570 }
571 push(e.node, e.key_byte, e.child_index, e.prefix);
572 }
573
575 void pop() noexcept {
576 UNODB_DETAIL_ASSERT(!empty());
577
578 const auto& e = top();
579 const auto n = static_cast<std::size_t>(
580 (!(art_policy::can_eliminate_leaf && e.packed_leaf) &&
581 e.node.type() != node_type::LEAF)
582 ? e.prefix.length() + 1
583 : 0);
584 keybuf_.pop(n);
585 stack_.pop();
586 }

1598template <typename Key, typename Value>
1599detail::node_ptr db<Key, Value>::build_chain(art_key_type k,
1600 detail::node_ptr child,
1601 tree_depth_type start_depth) {
1602 constexpr std::size_t cap = detail::key_prefix_capacity;
1603 const auto full_key = k.get_key_view();
1604 const auto key_len = k.size();
1605 const auto start = static_cast<std::size_t>(start_depth);
1606 auto current = child;
1607 bool child_is_value = art_policy::can_eliminate_leaf;
1608 bool owns_current =
1609 false; // set true once we've built at least one chain node
1610 // Build bottom-up: start from end of key, work toward start_depth.
1611 // Each chain I4 consumes up to cap prefix bytes + 1 dispatch byte.
1612 try {
1613 std::size_t pos = key_len;
1614 while (pos > start + cap) {
1615 const auto depth = pos - cap - 1;
1616 const auto dispatch = full_key[pos - 1];
1617 auto remaining = k;
1618 remaining.shift_right(depth);
1619 auto chain{
1620 inode_4::create(*this, full_key, remaining,
1621 tree_depth_type{static_cast<std::uint32_t>(depth)},
1622 dispatch, current)};
1623 if (child_is_value) {
1624 chain->set_value_bit(0);
1625 child_is_value = false;
1626 }
1627 current = detail::node_ptr{chain.release(), node_type::I4};
1628 owns_current = true;
1629#ifdef UNODB_DETAIL_WITH_STATS
1630 account_growing_inode<node_type::I4>();
1631#endif
1632 pos = depth;
1633 }
1634 // Tail: remaining bytes from start_depth to pos.
1635 if (pos > start) {
1636 const auto dispatch = full_key[pos - 1];
1637 auto chain{inode_4::create(
1638 *this, full_key, tree_depth_type{static_cast<std::uint32_t>(start)},
1639 static_cast<detail::key_prefix_size>(pos - start - 1), dispatch,
1640 current)};
1641 if (child_is_value) {
1642 chain->set_value_bit(0);
1643 }
1644 current = detail::node_ptr{chain.release(), node_type::I4};
1645 owns_current = true;
1646#ifdef UNODB_DETAIL_WITH_STATS
1647 account_growing_inode<node_type::I4>();
1648#endif
1649 }
1650 } catch (...) {
1651 // On failure, free whatever current points to. If owns_current is
1652 // true, current is a partial chain (I4 tree). If false, current is
1653 // the original child — a leaf node_ptr on the leaf path, or a packed
1654 // value on the VIS path. delete_subtree handles both I4 and LEAF.
1655 // For VIS packed values (!child_is_value is false only after the
1656 // first chain node is built), the caller's packed bits are not a
1657 // real pointer — but we only reach here with a packed value if
1658 // owns_current is false AND child_is_value is true, meaning no
1659 // chain was built and the packed value was never wrapped. In that
1660 // case we must NOT call delete_subtree on packed bits.
1661 if (owns_current) {
1662 art_policy::delete_subtree(current, *this);
1663 } else if (!child_is_value) {
1664 art_policy::delete_subtree(current, *this);
1665 }
1666 throw;
1667 }
1668 return current;
1669}

2310 //
2311 // Note: We are probing with a key byte which does not appear
2312 // in our list of keys (this was verified above) so this will
2313 // always be the index the first entry whose key byte is
2314 // greater-than the probe value and [false] if there is no
2315 // such entry.
2316 //
2317 // Note: [node] has not been pushed onto the stack yet!
2318 auto nxt = inode->gte_key_byte(node_type, remaining_key[0]);
2319 if (!nxt) {
2320 // Pop entries off the stack until we find one with a
2321 // right-sibling of the path we took to this node and then
2322 // do a left-most descent under that right-sibling. If there
2323 // is no such parent, we will wind up with an empty stack
2324 // (iterator at the end) and return that state.
2325 if (!empty()) pop();
2326 while (!empty()) {
2327 const auto& centry = top();
2328 const auto cnode{centry.node}; // possible parent from the stack
2329 auto* const icnode{cnode.template ptr<inode_type*>()};
2330 const auto cnxt = icnode->next(
2331 cnode.type(), centry.child_index); // right-sibling.
2332 if (cnxt) {
2333 const auto& e2 = cnxt.value();
2334 pop();
2335 push(e2);
2336 auto nchild = icnode->get_child(cnode.type(), e2.child_index);
2337 return descend_left(icnode, cnode.type(), e2.child_index, nchild);
2338 }
2339 pop();
2340 }
2341 return *this; // stack is empty (aka end iterator).
2342 }
2343 const auto& tmp = nxt.value(); // unwrap.
2344 const auto child_index = tmp.child_index;
2345 const auto child = inode->get_child(node_type, child_index);
2346 push(node, tmp.key_byte, child_index, tmp.prefix); // the path we took
2347 return descend_left(inode, node_type, child_index, child);
2348 }
2349 // REV: Take the prior child_index that is mapped and then do
2350 // a right-most descent to land on the key that is the
2351 // immediate predecessor of the desired key in the data.
2352 auto nxt = inode->lte_key_byte(node_type, remaining_key[0]);
2353 if (!nxt) {
2354 // Pop off the current entry until we find one with a
2355 // left-sibling and then do a right-most descent under that
2356 // left-sibling. In the extreme case there is no such
2357 // previous entry and we will wind up with an empty stack.
2358 if (!empty()) pop();
2359 while (!empty()) {
2360 const auto& centry = top();
2361 const auto cnode{centry.node}; // possible parent from stack
2362 auto* const icnode{cnode.template ptr<inode_type*>()};
2363 const auto cnxt =
2364 icnode->prior(cnode.type(), centry.child_index); // left-sibling.
2365 if (cnxt) {
2366 const auto& e2 = cnxt.value();
2367 pop();
2368 push(e2);
2369 auto nchild = icnode->get_child(cnode.type(), e2.child_index);
2370 return descend_right(icnode, cnode.type(), e2.child_index, nchild);
2371 }
2372 pop();
2373 }
2374 return *this; // stack is empty (aka end iterator).
2375 }
2376 const auto& tmp = nxt.value(); // unwrap.
2377 const auto child_index{tmp.child_index};
2378 const auto child = inode->get_child(node_type, child_index);
2379 push(node, tmp.key_byte, child_index, tmp.prefix); // the path we took
2380 return descend_right(inode, node_type, child_index, child);
2381 }
2382 // Simple case. There is a child for the current key byte.
2383 const auto child_index{res.first};
2384 const auto* const child{res.second};
2385 push(node, remaining_key[0], child_index, key_prefix);
2386 if constexpr (art_policy::can_eliminate_leaf) {
2387 if (inode->is_value_in_slot(node_type, child_index)) {
2388 push_leaf(*child);
2389 if (remaining_key.size() <= 1) {
2390 match = true;
2391 return *this;
2392 }
2393 // VIS key is a strict prefix of search key (VIS < search).
2394 return fwd ? next() : *this;
2395 }
2396 }
2397 node = *child;
2398 remaining_key.shift_right(1);
2399 } // while ( true )
2400 UNODB_DETAIL_CANNOT_HAPPEN();
2401}
