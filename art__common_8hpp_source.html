221inline void ensure_capacity(std::byte*& buf, size_t& cap, size_t off,
222 size_t min_capacity) {
223 // Find the smallest power of two >= min_capacity.
224 const auto asize = std::bit_ceil(min_capacity);
225 auto* tmp = detail::allocate_aligned(asize); // new allocation.
226 std::memcpy(tmp, buf, off); // copy over the data.
227 if (cap > INITIAL_BUFFER_CAPACITY) { // free old buffer iff allocated
228 detail::free_aligned(buf);
229 }
230 buf = static_cast<std::byte*>(tmp);
231 cap = asize;
232}

527 key_encoder& encode_text(std::span<const std::byte> text) {
528 // truncate view to at most maxlen bytes.
529 text = (text.size_bytes() > maxlen) ? text.subspan(0, maxlen) : text;
530 // normalize padding by stripping off any trailing [pad] bytes.
531 auto sz = text.size_bytes();
532 for (; sz > 0; sz--) {
533 if (text[sz - 1] != pad) break;
534 }
535 text = text.subspan(0, sz); // adjust span in case truncated.
536 // Ensure enough room for the text, the pad byte, and the
537 // run-length encoding of the pad byte.
538 ensure_available(sz + 1 + sizeof(size_type));
539 const auto padlen{static_cast<size_type>(maxlen - sz)};
540 append_bytes(text); // append bytes to the buffer.
541 encode(static_cast<std::uint8_t>(pad)); // encode as unsigned byte
542 encode(padlen); // logical run-length of the pad byte.
543 return *this;
544 }

647 key_decoder& decode(std::int8_t& v) noexcept {
648 std::uint8_t u;
649 decode(u);
650 v = static_cast<std::int8_t>(
651 (u >= msb8) ? u - msb8 : ~static_cast<std::uint8_t>(msb8 - 1U - u));
652 return *this;
653 }

658 std::uint16_t u;
659 decode(u);
660 v = static_cast<std::int16_t>(
661 (u >= msb16) ? u - msb16 : ~static_cast<std::uint16_t>(msb16 - 1U - u));
662 return *this;
663 }
664
667 key_decoder& decode(std::int32_t& v) noexcept {
668 constexpr auto one = static_cast<std::uint32_t>(1);
669 std::uint32_t u;
670 decode(u);
671 v = (u >= msb32) ? static_cast<int32_t>(u - msb32)
672 : -static_cast<int32_t>(msb32 - one - u) - 1;
673 return *this;
674 }
675
678 key_decoder& decode(std::int64_t& v) noexcept {
679 constexpr auto one = static_cast<std::uint64_t>(1);
680 std::uint64_t u;
681 decode(u);
682 v = (u >= msb64) ? static_cast<int64_t>(u - msb64)
683 : -static_cast<int64_t>(msb64 - one - u) - 1LL;
684 return *this;
685 }
