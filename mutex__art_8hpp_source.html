41 [[nodiscard]] get_result get_internal(art_key_type k) const noexcept {
42 std::unique_lock guard{mutex};
43 const auto db_get_result{db_.get_internal(k)};
44 if (!db_get_result) {
45 guard.unlock();
46 return std::make_pair(db_get_result, std::unique_lock<std::mutex>{});
47 }
48 return std::make_pair(db_get_result, std::move(guard));
49 }

269 [[nodiscard]] static bool key_found(const get_result& result) noexcept {
270#ifndef NDEBUG
271 const auto& lock{result.second};
272 // NOLINTNEXTLINE(readability-simplify-boolean-expr)
273 assert(!result.first || lock.owns_lock());
