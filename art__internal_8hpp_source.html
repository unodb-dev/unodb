48[[nodiscard, gnu::pure]] inline int compare(const void* a, const size_t alen,
49 const void* b,
50 const size_t blen) noexcept {
51 // TODO(thompsonbry) consider changing this over to std::span
52 // arguments and do not let the (ptr,len) pattern propagate
53 // outwards.
54 const auto shared_length = std::min(alen, blen);
55 auto ret = std::memcmp(a, b, shared_length);
56 ret = (ret != 0) ? ret : ((alen == blen) ? 0 : (alen < blen ? -1 : 1));
57 return ret;
58}

66 const unodb::key_view b) noexcept {
67 return compare(a.data(), a.size_bytes(), b.data(), b.size_bytes());
68}

112 [[nodiscard, gnu::const]] static UNODB_DETAIL_CONSTEXPR_NOT_MSVC KeyType
113 make_binary_comparable(KeyType k) noexcept {
114 static_assert(std::endian::native == std::endian::little,
115 "Big-endian support needs implementing");

117 }
118
119 public:
124 UNODB_DETAIL_CONSTEXPR_NOT_MSVC explicit basic_art_key(KeyType key_) noexcept
125 requires std::is_integral_v<KeyType>

136 [[nodiscard, gnu::pure]] constexpr int cmp(
137 basic_art_key<KeyType> key2) const noexcept {
138 if constexpr (std::is_same_v<KeyType, key_view>) {
139 return compare(key, key2.key);
140 } else {
141 return std::memcmp(&key, &key2.key, sizeof(KeyType));
142 }
143 }
144
148 [[nodiscard, gnu::pure]] constexpr int cmp(key_view key2) const noexcept {
149 if constexpr (std::is_same_v<KeyType, unodb::key_view>) {
150 // variable length keys
151 return compare(key, key2);
152 } else {
153 // fixed width keys
154 return compare(&key, sizeof(KeyType), key2.data(), key2.size_bytes());
155 }
156 }
157
161 [[nodiscard, gnu::pure]] constexpr auto operator[](
162 std::size_t index) const noexcept {
163 if constexpr (std::is_same_v<KeyType, key_view>) {
164 return key[index];
165 } else {
166 // The key_bytes[] provides access the different byte positions
167 // in the primitive type [Key key].
168 UNODB_DETAIL_ASSERT(index < sizeof(KeyType));
169 return key_bytes[index];
170 }
171 }

184 [[nodiscard, gnu::pure]] constexpr key_view get_key_view() const noexcept {
185 if constexpr (std::is_same_v<KeyType, key_view>) {
186 return key;
187 } else {
188 return key_view(reinterpret_cast<const std::byte*>(&key), sizeof(key));
189 }
190 }

199 [[nodiscard, gnu::pure]] constexpr std::uint64_t get_u64() const noexcept {
200 if constexpr (std::is_same_v<KeyType, key_view>) {
201 return unodb::detail::get_u64(key);
202 } else {
203 return static_cast<std::uint64_t>(key);
204 }
205 }

218 constexpr void shift_right(const std::size_t num_bytes) noexcept {
219 if constexpr (std::is_same_v<KeyType, key_view>) {
220 UNODB_DETAIL_ASSERT(num_bytes <= key.size_bytes());
221 key = key.subspan(num_bytes);
222 } else {
223 UNODB_DETAIL_ASSERT(num_bytes <= sizeof(KeyType));
224 key >>= (num_bytes * 8);
225 }
226 }
227
231 constexpr size_t size() const noexcept {
232 if constexpr (std::is_same_v<KeyType, unodb::key_view>) {
233 return key.size_bytes();
234 } else {
235 return sizeof(KeyType);
236 }

496 [[nodiscard, gnu::const]] static std::uintptr_t tag_ptr(

513 }
