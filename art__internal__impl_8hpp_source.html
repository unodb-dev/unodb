64template <bool Enabled, class Storage>
65struct value_bitmask_field {
66 Storage bits{};
67
68 [[nodiscard]] constexpr bool test(std::uint8_t i) const noexcept {
69 return (bits >> i) & 1U;
70 }
71 constexpr void set(std::uint8_t i) noexcept {
72 bits |= static_cast<Storage>(Storage{1} << i);
73 }
74 constexpr void clear(std::uint8_t i) noexcept {
75 bits &= static_cast<Storage>(~(Storage{1} << i));
76 }
78 constexpr void remove_at(std::uint8_t i) noexcept {
79 const auto above =
80 static_cast<Storage>(bits >> static_cast<unsigned>(i + 1));
81 const auto below = static_cast<Storage>(bits & ((Storage{1} << i) - 1));
82 bits = static_cast<Storage>(below | (above << i));
83 }
86 constexpr void insert_at(std::uint8_t i) noexcept {
87 const auto above = static_cast<Storage>(bits >> i);
88 const auto below = static_cast<Storage>(bits & ((Storage{1} << i) - 1));
89 bits =
90 static_cast<Storage>(below | (above << static_cast<unsigned>(i + 1)));
91 }
92};

99 [[nodiscard]] constexpr bool test(std::uint8_t i) const noexcept {
100 return (static_cast<unsigned>(bits[static_cast<std::size_t>(i) / 8]) >>
101 (static_cast<unsigned>(i) % 8U)) &
102 1U;

481template <typename Key, typename Value, template <typename, typename> class Db>
482[[nodiscard]] auto make_db_leaf_ptr(basic_art_key<Key> k, Value v,
483 Db<Key, Value>& db
484 UNODB_DETAIL_LIFETIMEBOUND) {
485 using db_type = Db<Key, Value>;
486 using header_type = typename db_type::header_type;
487 using leaf_type = basic_leaf<leaf_key_type<Key, Value>, header_type>;
488
489 if constexpr (!can_eliminate_key_in_leaf_v<Key, Value> &&
490 std::is_same_v<Key, key_view>) {
491 if (UNODB_DETAIL_UNLIKELY(k.size() > leaf_type::max_key_size)) {
492 throw std::length_error("Key length must fit in std::uint32_t");
493 }
494 }
495
496 // Serialize value to bytes for leaf storage.
497 value_view leaf_val_bytes;
498 UNODB_DETAIL_DISABLE_MSVC_WARNING(26496)
499 [[maybe_unused]] std::byte val_buf[sizeof(Value)]{};
500 UNODB_DETAIL_RESTORE_MSVC_WARNINGS()
501 if constexpr (std::is_same_v<Value, value_view>) {
502 leaf_val_bytes = v;
503 } else {
504 static_assert(std::is_trivially_copyable_v<Value>);
505 std::memcpy(val_buf, &v, sizeof(v));
506 leaf_val_bytes = value_view{val_buf, sizeof(v)};
507 }
508
509 if (UNODB_DETAIL_UNLIKELY(leaf_val_bytes.size_bytes() >
510 leaf_type::max_value_size)) {
511 throw std::length_error("Value length must fit in std::uint32_t");
512 }
513
514 std::size_t size;
515 if constexpr (can_eliminate_key_in_leaf_v<Key, Value>) {
516 size = leaf_type::compute_size(
517 static_cast<typename leaf_type::value_size_type>(
518 leaf_val_bytes.size_bytes()));
519 } else {
520 size = leaf_type::compute_size(
521 static_cast<typename leaf_type::key_size_type>(k.size()),
522 static_cast<typename leaf_type::value_size_type>(
523 leaf_val_bytes.size_bytes()));
524 }
525
526 auto* const leaf_mem = static_cast<std::byte*>(
527 allocate_aligned(size, alignment_for_new<leaf_type>()));
528
529#ifdef UNODB_DETAIL_WITH_STATS
530 db.increment_leaf_count(size);
531#endif // UNODB_DETAIL_WITH_STATS
532
533 UNODB_DETAIL_DISABLE_MSVC_WARNING(26402)
534 UNODB_DETAIL_DISABLE_MSVC_WARNING(26400)
535 UNODB_DETAIL_DISABLE_MSVC_WARNING(26496)
536 auto* const leaf = [&]() noexcept {
537 if constexpr (can_eliminate_key_in_leaf_v<Key, Value>) {
538 return new (leaf_mem) leaf_type{leaf_val_bytes};
539 } else {
540 return new (leaf_mem) leaf_type{k, leaf_val_bytes};
541 }
542 }();
543 UNODB_DETAIL_RESTORE_MSVC_WARNINGS()
544 UNODB_DETAIL_RESTORE_MSVC_WARNINGS()
545 return basic_db_leaf_unique_ptr<Key, Value, header_type, Db>{
546 leaf, basic_db_leaf_deleter<db_type>{db}};
547 UNODB_DETAIL_RESTORE_MSVC_WARNINGS()
548}
