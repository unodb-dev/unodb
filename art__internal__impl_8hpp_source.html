106// TODO(thompsonbry) Partial or no key in leaf. Once we template for the Value
107// type, we can optimize for u64 or smaller values, the leaf should be replaced
108// by the use of a variant {Value,node_ptr} entry in the inode along with a bit
109// mask to indicate for each position whether it is a leaf or a node. This
110// provides a significant optimization for secondary index use cases (tree
111// height is reduced by one, no small allocations for leaves, the key is no
112// longer explicitly stored, the key size is no longer stored, the value size is
113// no longer stored, etc.). However, we would still need to allocate a leaf for
114// the case where Value is a std::span. But this becomes a simple immutable
115// data structure which exists solely to wrap the Value.
116template <class Key, class Header>
117class [[nodiscard]] basic_leaf final : public Header {
118 public:
121 using key_size_type = unodb::key_size_type;
122
125 using value_size_type = unodb::value_size_type;
126
128 static constexpr std::size_t max_key_size =
129 std::numeric_limits<key_size_type>::max();
130
132 static constexpr std::size_t max_value_size =
133 std::numeric_limits<value_size_type>::max();
134
136 using art_key_type = basic_art_key<Key>;
137
138 UNODB_DETAIL_DISABLE_MSVC_WARNING(26485)
139
140 UNODB_DETAIL_DISABLE_MSVC_WARNING(26481)
145 constexpr basic_leaf(art_key_type k, value_view v) noexcept
146 : key_size{static_cast<key_size_type>(k.size())},
147 value_size{static_cast<value_size_type>(v.size())} {
148 // Note: Runtime checks are handled upstream of this by
149 // make_db_leaf_ptr().
150 UNODB_DETAIL_ASSERT(k.size() <= max_key_size);
151 UNODB_DETAIL_ASSERT(v.size() <= max_value_size);
152
153 const auto tmp{k.get_key_view()};
154 std::memcpy(data, tmp.data(), key_size); // store encoded key
155 if (!v.empty()) std::memcpy(data + key_size, v.data(), value_size);
156 }
157 UNODB_DETAIL_RESTORE_MSVC_WARNINGS()

163 // TODO(thompsonbry) : Partial or no key in leaf?
164 [[nodiscard, gnu::pure]] constexpr auto get_key() const noexcept {
165 if constexpr (std::is_same_v<Key, key_view>) {
166 return art_key_type{get_key_view()};

168 // Use memcpy since alignment is not guaranteed because the
169 // [key] is not an explicit part of the leaf data structure.
170 //
171 // TODO(thompsonbry) memory align leaf::data[0] to 8 bytes?
172 Key u{};
173 std::memcpy(&u, data, sizeof(u));
174 // Note: The encoded key is stored in the leaf. Since the
175 // art_key constructor encodes the key, we need to decode the
176 // key before calling the constructor.
177 return art_key_type{bswap(u)};
178 }
179 }

288template <typename Key, typename Value, template <typename, typename> class Db>
289[[nodiscard]] auto make_db_leaf_ptr(basic_art_key<Key> k, value_view v,
290 Db<Key, Value>& db
291 UNODB_DETAIL_LIFETIMEBOUND) {
292 using db_type = Db<Key, Value>;
293 using header_type = typename db_type::header_type;
294 using leaf_type = basic_leaf<Key, header_type>;
295
296 // TODO(thompsonbry) We should have a discussion about limits. To
297 // my mind, limits should be explicit configuration values, not
298 // uint32_t.
299 if constexpr (std::is_same_v<Key, key_view>) {
300 if (UNODB_DETAIL_UNLIKELY(k.size() > leaf_type::max_key_size)) {
301 throw std::length_error("Key length must fit in std::uint32_t");
302 }
303 }
304
305 if (UNODB_DETAIL_UNLIKELY(v.size_bytes() > leaf_type::max_value_size)) {
306 throw std::length_error("Value length must fit in std::uint32_t");
307 }
308
309 const auto size = leaf_type::compute_size(
310 static_cast<typename leaf_type::key_size_type>(k.size()),
311 static_cast<typename leaf_type::value_size_type>(v.size_bytes()));
312
313 auto* const leaf_mem = static_cast<std::byte*>(
314 allocate_aligned(size, alignment_for_new<leaf_type>()));
315
316#ifdef UNODB_DETAIL_WITH_STATS
317 db.increment_leaf_count(size);
318#endif // UNODB_DETAIL_WITH_STATS
319
320 return basic_db_leaf_unique_ptr<Key, Value, header_type, Db>{
321 new (leaf_mem) leaf_type{k, v}, basic_db_leaf_deleter<db_type>{db}};
322}

967 constexpr void cut(key_prefix_size cut_len) noexcept {
968 UNODB_DETAIL_ASSERT(cut_len > 0);
969 UNODB_DETAIL_ASSERT(cut_len <= length());
970
971 u64 = ((u64 >> (cut_len * 8)) & key_bytes_mask) |
972 length_to_word(static_cast<key_prefix_size>(length() - cut_len));
973

983 constexpr void prepend(const key_prefix& prefix1,
984 std::byte prefix2) noexcept {
985 UNODB_DETAIL_ASSERT(length() + prefix1.length() < key_prefix_capacity);
986
987 const auto prefix1_bit_length = prefix1.length() * 8U;
988 const auto prefix1_mask = (1ULL << prefix1_bit_length) - 1;
989 const auto prefix3_bit_length = length() * 8U;
990 const auto prefix3_mask = (1ULL << prefix3_bit_length) - 1;
991 const auto prefix3 = u64 & prefix3_mask;
992 const auto shifted_prefix3 = prefix3 << (prefix1_bit_length + 8U);
993 const auto shifted_prefix2 = static_cast<std::uint64_t>(prefix2)
994 << prefix1_bit_length;
995 const auto masked_prefix1 = prefix1.u64 & prefix1_mask;
996
997 u64 = shifted_prefix3 | shifted_prefix2 | masked_prefix1 |
998 length_to_word(length() + prefix1.length() + 1U);
999
1000 UNODB_DETAIL_ASSERT(f.key_prefix_length.load() <= key_prefix_capacity);
1001 }

2111 [[nodiscard, gnu::pure]] find_result find_child(std::byte key_byte) noexcept {
2112#ifdef UNODB_DETAIL_X86_64
2113 const auto replicated_search_key =
2114 _mm_set1_epi8(static_cast<char>(key_byte));
2115 const auto keys_in_sse_reg =
2116 _mm_cvtsi32_si128(static_cast<std::int32_t>(keys.integer.load()));

2118 _mm_cmpeq_epi8(replicated_search_key, keys_in_sse_reg);

2121 static_cast<unsigned>(_mm_movemask_epi8(matching_key_positions)) & mask;

2132 // NOLINTNEXTLINE(misc-const-correctness)

2135 const auto matching_key_positions =
2136 vceqq_u8(replicated_search_key, keys_in_neon_reg);
2137 const auto u64_pos_in_vec =
2138 vget_low_u64(vreinterpretq_u64_u8(matching_key_positions));
2139 // NOLINTNEXTLINE(misc-const-correctness)
2140 const auto pos_in_scalar = vget_lane_u64(u64_pos_in_vec, 0);

2148#else // #ifdef UNODB_DETAIL_X86_64
2149 // Bit twiddling:
2150 // contains_byte: std::countr_zero: for key index:

2153 // 0x8000 0x0F 1
2154 // 0x80 0x07 0

2160 static_cast<typename decltype(keys.byte_array)::size_type>(
2161 std::countr_zero(match) >> 3U);
